CREATE DATABASE TiendaOnline;
USE TiendaOnline;

-- Tabla de Usuarios
CREATE TABLE Usuarios (
    id_usuario INT AUTO_INCREMENT PRIMARY KEY,
    FOTO LONGBLOB,
    nombre VARCHAR(100) NOT NULL,
    email VARCHAR(100) UNIQUE NOT NULL,
    password VARCHAR(255) NOT NULL,
    direccion VARCHAR(255) NOT NULL,
    telefono VARCHAR(255) NOT NULL,
    fecha_registro DATETIME DEFAULT CURRENT_TIMESTAMP,
    rol ENUM('user', 'admin') NOT NULL DEFAULT 'user'
);


-- Tabla de Categorías
CREATE TABLE Categorias (
    id_categoria INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(100) UNIQUE NOT NULL,
    descripcion TEXT
);

-- Tabla de Cartas
CREATE TABLE cartas (
    id_carta INT PRIMARY KEY AUTO_INCREMENT,
    nombre VARCHAR(255),
    stock INT,
    precio DECIMAL(10,2),
    scryfall_id VARCHAR(255),
    set_code VARCHAR(10),
    collector_number VARCHAR(20)
);

-- Tabla de Productos (sin id_categoria directamente)
CREATE TABLE productos (
    id_producto INT AUTO_INCREMENT PRIMARY KEY,
    nombre VARCHAR(255) NOT NULL,
    descripcion TEXT,
    precio DECIMAL(10,2) NOT NULL,
    stock INT NOT NULL,
    imagenes LONGTEXT
);

-- Tabla intermedia: muchos a muchos entre productos y categorías
CREATE TABLE ProductoCategoria (
    id_producto INT,
    id_categoria INT,
    PRIMARY KEY (id_producto, id_categoria),
    FOREIGN KEY (id_producto) REFERENCES Productos(id_producto) ON DELETE CASCADE,
    FOREIGN KEY (id_categoria) REFERENCES Categorias(id_categoria) ON DELETE CASCADE
);

-- Tabla de Pedidos
CREATE TABLE Pedidos (
    id_pedido INT AUTO_INCREMENT PRIMARY KEY,
    id_usuario INT,
    fecha_pedido TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    total DECIMAL(10,2) NOT NULL,
    estado ENUM('Pendiente', 'Pagado', 'Entregado', 'Cancelado') DEFAULT 'Pendiente',
    FOREIGN KEY (id_usuario) REFERENCES Usuarios(id_usuario) ON DELETE CASCADE
);

-- Tabla de Detalles de Pedido (para cartas y productos)
CREATE TABLE DetallePedido (
    id_detalle INT AUTO_INCREMENT PRIMARY KEY,
    id_pedido INT,
    tipo_item ENUM('producto', 'carta') NOT NULL,
    id_item INT NOT NULL,
    cantidad INT NOT NULL,
    precio DECIMAL(10,2) NOT NULL,
    FOREIGN KEY (id_pedido) REFERENCES Pedidos(id_pedido) ON DELETE CASCADE
);

-- Tabla de Deseados (wishlist)
CREATE TABLE Deseados (
    id_deseado INT AUTO_INCREMENT PRIMARY KEY,
    id_usuario INT NOT NULL,
    id_carta INT DEFAULT NULL,
    id_producto INT DEFAULT NULL,

    -- Restricciones para evitar duplicados
    UNIQUE (id_usuario, id_carta),
    UNIQUE (id_usuario, id_producto),

    -- Relaciones
    FOREIGN KEY (id_usuario) REFERENCES Usuarios(id_usuario) ON DELETE CASCADE,
    FOREIGN KEY (id_carta) REFERENCES cartas(id_carta) ON DELETE SET NULL,
    FOREIGN KEY (id_producto) REFERENCES Productos(id_producto) ON DELETE SET NULL
);

-- Tabla de Carrito
CREATE TABLE Carrito (
    id_carrito INT AUTO_INCREMENT PRIMARY KEY,
    id_usuario INT NOT NULL,
    tipo_item ENUM('producto', 'carta') NOT NULL,
    id_item INT NOT NULL,
    cantidad INT NOT NULL,
    fecha_agregado TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (id_usuario) REFERENCES Usuarios(id_usuario) ON DELETE CASCADE
);
